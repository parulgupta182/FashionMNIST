def predict(model,image):

  accuracy = max(model.predict(image.reshape(1,784))[0])*100
  pc = model.predict(image.reshape(1,784)).argmax(-1)[0]
  name = class_names[pc]

  print(pc,name," accuracy:", accuracy,"%")
  plt.imshow(image,cmap=plt.cm.binary)
  plt.show()
